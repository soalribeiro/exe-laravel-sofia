    const url = new URL("http://localhost/api/noticia");

    "Content-Type": "application/json",
    "Accept": "application/json",
}

    "titulo-jor": "odit",
    "corpo-jor": "et",
    "image": "ullam",
    "jornal_id": 1,
    "seccao_id": 19
}

fetch(url, {
    method: "POST",
    headers: headers,
    body: body

    .then(json => console.log(json));
